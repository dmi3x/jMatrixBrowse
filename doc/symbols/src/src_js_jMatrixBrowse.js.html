127     function checkOverflow(element, container) {
128       var containerOffset = container.offset();
129       var elementOffset = element.offset();
130       
131       var top = elementOffset.top - containerOffset.top;
132       var left = elementOffset.left - containerOffset.left;
133       var width = element.width();
134       var height = element.height();
135 
136       if (left > container.width()) {
137         return OVERFLOW_RIGHT;
138       }
139       
140       if (top > container.height()) {
141         return OVERFLOW_BOTTOM;
142       }
143       
144       if (top+height < 0) {
145         return OVERFLOW_TOP;
146       }
147       
148       if (left+width < 0) {
149         return OVERFLOW_LEFT;
150       }
151       
152       return OVERFLOW_NONE;
153     }

287     function checkAndRepositionCells() {
288       // TODO: We might need to check more elements in case of a quick drag.
289       
290       // Row on top might overflow from the top.
291       checkAndRepositionCellRow(_cellElements, _container, 0, OVERFLOW_TOP);
292       // Row on bottom might overflow from the bottom.
293       checkAndRepositionCellRow(_cellElements, _container, _cellElements.length-1, OVERFLOW_BOTTOM);
294       // Column on left might overflow from the left.
295       checkAndRepositionCellCol(_cellElements, _container, 0, OVERFLOW_LEFT);
296       // Column on right might overflow from the right.
297       checkAndRepositionCellCol(_cellElements, _container, _cellElements[0].length-1, OVERFLOW_RIGHT);

300     /**
301      * Checks if the given row is overflowing and repositions if necessary. 
302      * @param {matrix} cellElements - The matrix of DOM objects representing cells.
303      * @param {jQuery Object} container - The container against which to check the overflow.
304      * @param {Number} row - Index of row to check the overflow for.
305      * @param {Number} overflow - Type of the overflow to check for.
306      */
307     function checkAndRepositionCellRow(cellElements, container, row, overflow) {
308       if (cellElements[row].length > 0 && isOverflowing(jQuery(cellElements[row][0]), container, overflow)) {
309         
310         var cellRow;
311         switch (overflow) {
312           case OVERFLOW_TOP:
313             // The row is overflowing from top. Move it to bottom. 
314             var height = cellElements.length;
315             var lastCell = (cellElements[height-1].length > 0) ? jQuery(cellElements[height-1][0]) : undefined;
316             if (lastCell === undefined) {
317               console.error('Unable to resposition row ' + row + ' overflowing from top.')
318               return;
319             }
320             // Change the position of all elements in the row.
321             var newTop = lastCell.position().top + lastCell.height();
322             for (var i = 0, w = cellElements[row].length; i < w; ++i) {
323               jQuery(cellElements[row][i]).css({
324                 top: newTop
325               });
326             }
327             
328             // Move row in matrix to end
329             cellRow = cellElements.splice(row,1); // Remove row at [row]
330             if (cellRow.length > 0)
331               cellElements.push(cellRow[0]);  // Insert row at the end.
332             
333             break;
334 
335           case OVERFLOW_BOTTOM:
336             // The row is overflowing from bottom. Move it to top.
337             var firstCell = (cellElements.length > 0 && cellElements[0].length > 0)?jQuery(cellElements[0][0]):undefined;
338             if (firstCell === undefined) {
339               console.error('Unable to resposition row ' + row + ' overflowing from bottom.')
340               return;
341             }
342             // Change the position of all elements in the row.
343             var newBottom = firstCell.position().top;
344             for (var i = 0, w = cellElements[row].length; i < w; ++i) {
345               jQuery(cellElements[row][i]).css({
346                 top: newBottom - jQuery(cellElements[row][i]).height()
347               });
348             }
349             // Move row in matrix to first
350             cellRow = cellElements.splice(row,1);  // Remove row at [row]
351             if (cellRow.length > 0)
352               cellElements.splice(0,0,cellRow[0]);  // Insert row at the beginning.
353             
354             break;
355         }
356       }
357     }

359     /**
360      * Checks if the given column is overflowing and repositions if necessary. 
361      * @param {matrix} cellElements - The matrix of DOM objects representing cells.
362      * @param {jQuery Object} container - The container against which to check the overflow.
363      * @param {Number} col - Index of column to check the overflow for.
364      * @param {Number} overflow - Type of the overflow to check for.
365      */
366     function checkAndRepositionCellCol(cellElements, container, col, overflow) {
367       if (isOverflowing(jQuery(cellElements[0][col]), container, overflow)) {
368         // TODO: Do error checks.
369         switch (overflow) {
370           case OVERFLOW_LEFT:
371             // The row is overflowing from left. Move it to right. 
372             if (cellElements.length <= 0 || cellElements[0].length <= 0) {
373               console.error('Unable to resposition col ' + col + ' overflowing from left.');
374               return;
375             }
376             // Change the position of all elements in the column.
377             var w = cellElements[0].length;
378             var lastCell = jQuery(cellElements[0][w-1]);
379             var newLeft = lastCell.position().left + lastCell.width();
380             for (var i = 0, h = cellElements.length; i < h; ++i) {
381               jQuery(cellElements[i][col]).css({
382                 left: newLeft
383               });
384             }
385             // Move col to end in matrix.
386             for (var i = 0, h = cellElements.length; i < h; ++i) {
387               var cell = cellElements[i].splice(col, 1); // Remove element at [i][col]
388               cellElements[i].push(cell[0]); // Insert element at end of row i
389             }
390             break;
391 
392           case OVERFLOW_RIGHT:
393             // The row is overflowing from right. Move it to left. 
394             if (cellElements.length <= 0 || cellElements[0].length <= 0) {
395               console.error('Unable to resposition col ' + col + ' overflowing from left.');
396               return;
397             }
398             var firstCell = jQuery(cellElements[0][0]);
399             // Change the position of all elements in the column.
400             var newRight = firstCell.position().left;
401             for (var i = 0, h = cellElements.length; i < h; ++i) {
402               jQuery(cellElements[i][col]).css({
403                 left: newRight - jQuery(cellElements[i][col]).width()
404               });
405             }
406             // Move col to first in matrix.
407             for (var i = 0, h = cellElements.length; i < h; ++i) {
408               var cell = cellElements[i].splice(col, 1); // Remove element at [i][col]
409               cellElements[i].splice(0,0,cell[0]); // Insert element to [i][0]
410             }
411             break;
412         }
413       }
414     }

416     /**
417      * Check if the given cell is overflowing the container dimensions. 
418      * @param {jQuery Object} element - jQuery object for the element to be checked for overflow.
419      * @param {jQuery Object} container - The container against which to check the overflow.
420      * @param {Number} overflow - Type of oevrflow to check.
421      * @returns true if there is an overflow. false otherwise.
422      */
423     function isOverflowing(element, container, overflow) {
424       var containerOffset = container.offset();
425       var elementOffset = element.offset();
426       
427       var top = elementOffset.top - containerOffset.top;
428       var left = elementOffset.left - containerOffset.left;
429       var width = element.width();
430       var height = element.height();
431 
432       switch (overflow) {
433         case OVERFLOW_LEFT: 
434           return (left+width < 0);
435         case OVERFLOW_RIGHT: 
436           return (left > container.width());
437         case OVERFLOW_TOP: 
438           return (top+height < 0);
439         case OVERFLOW_BOTTOM: 
440           return (top > container.height());
441       }
442       return false;
443     }

445     /**
446      * Creates an empty matrix with size obtained from API and appends to content.
447      * @param {jQuery object} container - The element that acts as the matrix container (element that invoked jMatrixBrowse).
448      * @returns {jQuery object} content where matrix is generated.
449      */
450     function generateInitialMatrixContent(container) {
451       var size = getMatrixSize();
452       if (size == undefined) {
453         console.error("Unable to get matrix size");
454         return;
455       }
456       
457       var windowSize = getWindowSize();
458       if (windowSize == undefined) {
459         console.error("Unable to get window size");
460         return;
461       }
462       
463       var content = createContentDiv(container);
464       content.css({
465         'position' : 'absolute',
466         'top' : 0,
467         'left' : 0
468       });
469       
470       var cellHeight = _container.height()/windowSize.height;
471       var cellWidth = _container.width()/windowSize.width;
472       
473       var windowPosition = getWindowPosition();
474       
475       _cellElements = [];
476       // Generate matrix content for only the rows that are in the window.
477       var frag = document.createDocumentFragment();
478       for (var row=windowPosition.row; row < windowPosition.row+windowSize.height; row++) {
479         _cellElements.push([]);
480         for (var col=windowPosition.col; col < windowPosition.col + windowSize.width; col++) {
481           var elem = document.createElement("div");
482           elem.style.backgroundColor = row%2 + col%2 > 0 ? "#ddd" : "";
483           elem.style.width = cellWidth + "px";
484           elem.style.height = cellHeight + "px";
485           elem.style.position = "absolute";
486           elem.style.top = (row-windowPosition.row)*cellHeight + "px";
487           elem.style.left = (col-windowPosition.col)*cellWidth + "px";
488           elem.style.display = "inline-block";
489           elem.style.textIndent = "6px";
490           elem.innerHTML = row + "," + col;
491           elem.className += " jMatrixBrowse-cell " + generateClassNameForCell(row, col);
492           frag.appendChild(elem);
493           _cellElements[row-windowPosition.row].push(elem);
494         }
495       }
496       content.append(frag);
497       return content;
498     }

500     /**
501      * Initialize the jMatrixBrowse.
502      * @param {jQuery object} elem - the element to which to attach the jMatrixBrowse.
503      */
504     function init(elem) {
505       // Get user options
506       var options = getUserOptions(elem);
507 
508       // Extending user options with application defaults
509       var settings = extendDefaults(options);
510       setSettings(settings);
511 
512       // Initialize mock api
513       initApi('test');
514 
515       _container = elem;
516       
517       // Create draggable area and add matrix to it.
518       _dragContainer = createDragContainer(_container);
519       
520       // Generate initial content
521       _content = generateInitialMatrixContent(_dragContainer);
522       
523       // Scroll to the initial position
524       var windowPosition = getWindowPosition();
525       _self.scrollTo(windowPosition.row, windowPosition.col);
526       
527       // Load data
528       _self.reloadData();
529     }

559     /**
560      * Reload data in the matrix for the visible window. 
561      */
562     this.reloadData = function() {
563       var cellWindow = getCellWindow(_currentCell);
564       if (cellWindow == undefined) {
565         console.error('Unable to get cell window.');
566         return;
567       }
568       var response = _api.getResponse(cellWindow);
569       
570       if (response && response.data) {
571         for (var i = 0; i < response.data.length; ++i) {
572           for (var j = 0; j < response.data[i].length; ++j) {
573             // TODO: If we support named methods, the data should be extracted for the named method corresponding to current layer.
574             var cellData = response.data[i][j]; 
575             var row = cellWindow.row1 + i;
576             var col = cellWindow.col1 + j;
577             jQuery('.' + generateClassNameForCell(row, col)).html(cellData);  
578           }
579         }
580       }
