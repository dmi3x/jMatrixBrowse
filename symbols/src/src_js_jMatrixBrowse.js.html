 15 /**

 17  * @name fn
 18  * @class

 23 (function( jQuery ) {
 24   /**
 25    * jMatrixBrowse - a jQuery plugin to create large draggable(like Google Maps)
 26    * matrices. 
 27    *
 28    * @class jMatrixBrowse
 29    * @memberOf jQuery.fn
 30    */
 31   jQuery.fn.jMatrixBrowse = function() {
 32 
 33     var _self = this;
 34     
 35     var _settings;     // user Settings
 36     var _api;          // API object
 37     var _container;    // container for jMatrixBrowse
 38     var _content;      // content of jMatrixBrowse
 39     var _scroller;     // scroller object used for browsing
 40     var _cellPosition; // cell position for the component
 41     var _currentCell;  // currently shown cell (TODO: for now this cell is on left corner of matrix)
 42 
 43     /**
 44      * Initialize the API
 45      * @param {String} type type of api: 'test' initializes the mockAPI
 46      */
 47     function initApi(type) {
 48       if (type === 'test') {
 49         _api = new MockApi();
 50       } else {
 51         console.error('API ' + type + 'not yet supported.');
 52       }
 53     }
 54 
 55     /**
 56      * Get matrix size from api. 
 57      * @returns {Object} size - size of the matrix. 
 58      * @returns {Number} size.width - width of the matrix. 
 59      * @returns {Number} size.height - height of the matrix.
 60      */
 61     function getMatrixSize() {
 62       var response = _api.getResponse({
 63         'row1': 0,
 64         'col1': 0,
 65         'row2': 0,
 66         'col2': 0
 67       });
 68       if (response)
 69         return response.matrix;
 70     }
 71 
 72     /** 
 73      * Get window size from settings. 
 74      * @returns {Object} size - size of the window. 
 75      * @returns {Number} size.width - width of the window. 
 76      * @returns {Number} size.height - height of the window.
 77      */
 78     function getWindowSize() {
 79       if (_settings && _settings.str_initialWindowSize) {
 80         var  position = parsePosition(_settings.str_initialWindowSize);
 81         return {
 82           height: position.row,
 83           width: position.col
 84         };
 85       }
 86     }
 87     
 88     /** 
 89      * Get position of window. 
 90      * @returns {Object} position - position of the top-left corner of window. 
 91      * @returns {Number} position.row - row index of the position.
 92      * @returns {Number} position.col - column index of the position.
 93      */
 94     function getWindowPosition() {
 95       if (_settings && _settings.str_initialWindowPosition) {
 96         var  position = parsePosition(_settings.str_initialWindowPosition);
 97         return position;
 98       }
 99     }
100     
101     /** 
102      * Utility function that parses a position (row,col).
103      * @returns {Object} position - Object representing the position denoted by the string. 
104      * @returns {Number} position.row - row index of the position.
105      * @returns {Number} position.col - column index of the position.
106      */
107     function parsePosition(str_position) {
108       var arr_position = str_position.split(',');
109       if (arr_position.length == 2) {
110         return {
111           row: parseInt(arr_position[0]),
112           col: parseInt(arr_position[1])
113         };
114       }
115     }
116     
117     /**
118      * Get user defined options from data-* elements.
119      * @param {jQuery object} elem - the element to retrieve the user options from. 
120      * @returns {Object} options - User's options for the plugin. 
121      * @returns {boolean} options.boo_jMatrixBrowser - Is jMatrixBrowse active for the container.
122      * @returns {string} options.str_api - URI for the API.
123      * @returns {string} options.str_initialWindowSize - comma separated (TODO: no checks performed) window size as (width, height).
124      * @returns {string} options.str_initialWindowPosition - comma separated (TODO: no checks performed) window position as (row,col).
125      */
126     function getUserOptions(elem) {
127       var options = {
128         boo_jMatrixBrowser: (elem.attr('data-jmatrix_browser') === 'true'),
129         str_api: elem.attr('data-api'),
130         str_initialWindowSize: elem.attr('data-initial-window-size'),
131         str_initialWindowPosition: elem.attr('data-initial-window-position')
132       };
133       return options;
134     }
135 
136     /**
137      * Set the settings object.
138      * @param {Object} settings
139      */
140     function setSettings(settings) {
141       _settings = settings;
142     }
143 
144     /**
145      * Get settings object.
146      */
147     function getSettings() {
148       return _settings;
149     }
150 
151     /**
152      * Extend the user's settings with defaults.
153      * @returns {Object} options - User's options for the plugin. 
154      * @returns {boolean} options.boo_jMatrixBrowser - Is jMatrixBrowse active for the container.
155      * @returns {string} options.str_api - URI for the API.
156      * @returns {string} options.str_initialWindowSize - comma separated (TODO: no checks performed) window size as (width, height).
157      * @returns {string} options.str_initialWindowPosition - comma separated (TODO: no checks performed) window position as (row,col).
158      */
159     function extendDefaults(options) {
160       return jQuery.extend({
161         str_initialWindowPosition: '0,0',
162         str_initialWindowSize: '10,10'
163       }, options);
164     }
165 
166     /**
167      * Generate class name for given cell position.
168      * @param {Number} row - zero indexed row of the element.
169      * @param {Number} col - zero indexed column of the element.
170      * @returns {string} className - class name for the cell element.
171      */
172     function generateClassNameForCell(row, col) {
173       return "j-matrix-browse-cell-" + "row" + row + "col" + col;
174     }
175     
176     /**
177      * Get the window end points which has given point at its top left corner. 
178      * @param {Object} position - position of the cell.
179      * @param {Number} position.row - row of the cell.
180      * @param {Number} position.col - column of the cell.
181      * @returns {Object} window - Object representing the window coordinates. 
182      * @returns {Number} window.row1 - row index of the top left corner.
183      * @returns {Number} window.col1 - column index of the top left corner.
184      * @returns {Number} window.row2 - row index of the bottom right corner.
185      * @returns {Number} window.col2 - column index of the bottom right corner.
186      */
187     function getCellWindow(position) {
188       var size = getMatrixSize();
189       if (size == undefined) {
190         console.error("Unable to get matrix size");
191         return;
192       }
193       
194       var windowSize = getWindowSize();
195       if (windowSize == undefined) {
196         console.error("Unable to get window size");
197         return;
198       }
199       
200       return obj_cellWindow = {
201         row1: position.row,
202         col1: position.col,
203         row2: Math.min(position.row + windowSize.height, size.height),
204         col2: Math.min(position.col + windowSize.width, size.width) 
205       };
206     }
207     
208     /**
209      * Creates an empty matrix with size obtained from API and appends to content.
210      * @param {jQuery object} content - The element that acts as the matrix content.
211      * @param {jQuery object} container - The element that acts as the matrix container (parent of content).
212      */
213     function generateMatrixInDom(content, container) {
214 
215       var size = getMatrixSize();
216       if (size == undefined) {
217         console.error("Unable to get matrix size");
218         return;
219       }
220       
221       var windowSize = getWindowSize();
222       if (windowSize == undefined) {
223         console.error("Unable to get window size");
224         return;
225       }
226       
227       var w = container.width();
228       var h = container.height();
229       var cellWidth = Math.round(container.width()/windowSize.width);
230       var cellHeight = Math.round(container.height()/windowSize.height);
231       
232       content.width(cellWidth*size.width);
233       content.height(cellHeight*size.height);
234       
235       var windowPosition = getWindowPosition();
236       
237       // TODO: Idea: Generate only the visible part of matrix. 
238       // Still set the content height and width of content so 
239       // that Zynga scroller scrolls the other parts 
240       
241       // TODO: Remove hardcoded variables
242       // Generate matrix content
243       var frag = document.createDocumentFragment();
244       for (var row=windowPosition.row; row <= windowPosition.row+windowSize.height; row++) {
245         for (var col=windowPosition.col; col <= windowPosition.col + windowSize.width; col++) {
246           elem = document.createElement("div");
247           elem.style.backgroundColor = row%2 + col%2 > 0 ? "#ddd" : "";
248           elem.style.width = cellWidth + "px";
249           elem.style.height = cellHeight + "px";
250           elem.style.display = "inline-block";
251           elem.style.textIndent = "6px";
252           elem.innerHTML = row + "," + col;
253           elem.className += " " + generateClassNameForCell(row, col);
254           frag.appendChild(elem);
255         }
256         elem = document.createElement("br");
257         frag.appendChild(elem);
258       }
259       content.append(frag);
260     }
261 
262     /**
263      * Initialize the jMatrixBrowse.
264      * @param {jQuery object} elem - the element to which to attach the jMatrixBrowse.
265      */
266     function init(elem) {
267       // Get user options
268       var options = getUserOptions(elem);
269 
270       // Extending user options with application defaults
271       var settings = extendDefaults(options);
272       setSettings(settings);
273 
274       // Initialize mock api
275       initApi('test');
276 
277       _container = elem;
278       _content = jQuery(document.createElement('div'));
279       _container.append(_content);
280       
281       // Generate matrix content and add to DOM
282       generateMatrixInDom(_content, _container);
283 
284       // Attach EasyScroller to elem
285       _scroller = new EasyScroller(_content[0]);
286       
287       // TODO: Use matrix data instead of indices generated in previous step.
288       
289       // Scroll to the initial position
290       var windowPosition = getWindowPosition();
291       _self.scrollTo(windowPosition.row, windowPosition.col);
292 
293       // Load data
294       _self.reloadData();
295     }
296 
297     //Public API
298     
299     //TODO: Might not work when more than one jMatrixBrowse on the same page. 
300     /**
301      * Get the cell position for cell at (row,col).
302      * @param {Number} row - row index of the cell.
303      * @param {Number} col - column index of the cell.
304      * @returns {Object} position - position of the cell. 
305      * @returns {Number} position.top - top coordinate of the cell. 
306      * @returns {Number} position.left - left coordinate of the cell. 
307      */
308     this.getCellPosition = function (row, col) {
309       return jQuery('.' + generateClassNameForCell(row,col)).position();
310     };
311     
312     /**
313      * Scroll to given position. 
314      * @param {Number} row - row index of the cell.
315      * @param {Number} col - column index of the cell.
316      */
317     this.scrollTo = function (row, col) {
318       _cellPosition = _self.getCellPosition(row, col);
319       _currentCell = {
320         row: row,
321         col: col
322       };
323       _scroller.scroller.scrollTo(_cellPosition.left, _cellPosition.top);
324     };
325     
326     /**
327      * Reload data in the matrix for the visible window. 
328      */
329     this.reloadData = function() {
330       var cellWindow = getCellWindow(_currentCell);
331       if (cellWindow == undefined) {
332         console.error('Unable to get cell window.');
333         return;
334       }
335       var response = _api.getResponse(cellWindow);
336       
337       if (response && response.data) {
338         for (var i = 0; i < response.data.length; ++i) {
339           for (var j = 0; j < response.data[i].length; ++j) {
340             var cellData = response.data[i][j]; // TODO: If we support named methods, the data should be extracted for the named method corresponding to current layer.
341             var row = cellWindow.row1 + i;
342             var col = cellWindow.col1 + j;
343             jQuery('.' + generateClassNameForCell(row, col)).html(cellData);  
344           }
345         }
346       }
347     }
348     
349     /**
350      * Main plugin function
351      */
352     jQuery('[data-jmatrix_browser=true]').each( function() {
353       init(jQuery(this));
354     });
355   };
